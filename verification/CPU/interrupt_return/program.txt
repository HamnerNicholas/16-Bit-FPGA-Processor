.ivt INT3 custom_label

; ------------------------------------------------------------
; Main program
; ------------------------------------------------------------

addi r0 1
copy r1

; INT3 is asserted while PC points here.
addi r1 1

; Execution resumes here after rint.
copy r1

addi r0 0x55
copy r2

halt


; ------------------------------------------------------------
; Interrupt service routine
; ------------------------------------------------------------

: custom_label

; Preserve interrupted accumulator value (0x02).
ssrf r7

addi r0 0xAA
copy r3

; Restore interrupted accumulator value.
rsrf r7

rint
